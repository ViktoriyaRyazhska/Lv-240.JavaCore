package HomeWork4;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Task3 {

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

		System.out.println("Please, input any 5 numbers from paragraph");
		int input = Integer.parseInt(br.readLine());
		int input2 = Integer.parseInt(br.readLine());
		int input3 = Integer.parseInt(br.readLine());
		int input4 = Integer.parseInt(br.readLine());
		int input5 = Integer.parseInt(br.readLine());
		
		int numbers[] = { input, input2, input3, input4, input5 };

		int min = numbers[0];
		int positive = 0;
		int n = 0;

		for (int i = 0; i < numbers.length; i++) {
			if (numbers[i] < min) {
				min = numbers[i];
				n = i;
			}
			if (numbers[i] > 0) {
				positive++;
			}
			if (positive == 2) {
				System.out.println("The second positive number is on " + (i + 1) + " position");
				positive++;
			}
		}
		if (positive < 2) {
			System.out.println("There are only one or no positive numbers");
		}
		System.out.println(
				"Minimum number is " + min + " and it is on the " + (n + 1) + " position.");
	}

}

package HomeWork4;

import java.util.Scanner;

public class Task4 {

	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);

		int n = 0;
		int product = 1;

		do {
			System.out.println("Please enter any number");
			n = Integer.parseInt(sc.nextLine());
			if (n < 0) {
				System.out.println("Negative number " + n + " was entered");
				break;
			} else {
				if (n % 2 == 0) {
					product = product * n;
				}
			}
		} while (n >= 0);

		if (product > 1) {
			System.out.println("The product of all even numbers that were entered is " + product);
		} else {
			System.out.println("No positive even numbers were entered");
		}
		sc.close();
	}

}
