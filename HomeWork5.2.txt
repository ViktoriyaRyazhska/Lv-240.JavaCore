package HomeWork5;

public interface task2 {

	public double calculatePay();

}
/////////////////////////////

package HomeWork5;

abstract public class Employee implements task2 {
	public static int employeeId = 1;
	private String employeeName;

	public String getEmployeeName() {
		return employeeName;

	}

	public void setEmployeeName(String employeeName) {
		this.employeeName = employeeName;
	}

	public int getEmployeeId() {
		return employeeId++;
	}

	Employee(String employeeName) {
		this.employeeName = employeeName;

	}

	@Override
	public String toString() {
		return "Employee ID = " + getEmployeeId() + ", Employee Name = " + employeeName + " Average monthly Payment = "
				+ calculatePay() + " Euro";

	}

}

////////////////////////////////////////

package HomeWork5;

public class SalariedEmployee extends Employee implements task2 {

	private double hoursOfWork;
	private int socialSecurityNumber;
	private double hourly_rate;

	public SalariedEmployee(String employeeName, double hoursOfWork, double hourly_rate, int socialSecurityNumber) {
		super(employeeName);
		this.hoursOfWork = hoursOfWork;
		this.hourly_rate = hourly_rate;
		this.socialSecurityNumber = socialSecurityNumber;
	}

	public Integer getSocialSecurityNumber() {
		return socialSecurityNumber;
	}

	public void setSocialSecurityNumber(int socialSecurityNumber) {
		this.socialSecurityNumber = socialSecurityNumber;
	}

	@Override
	public double calculatePay() {
		return hoursOfWork * hourly_rate;
	}
}

package HomeWork5;

public class ContractEmployee extends Employee implements task2 {

	private double monthly_payment;
	private int federalTaxIdmember;

	public ContractEmployee(String employeeName, double monthly_payment, int federalTaxIdmember) {
		super(employeeName);
		this.monthly_payment = monthly_payment;
		this.federalTaxIdmember = federalTaxIdmember;
	}

	@Override
	public double calculatePay() {
		return monthly_payment;

	}

	public int getFederalTaxIdmember() {
		return federalTaxIdmember;
	}

	public void setFederalTaxIdmember(int federalTaxIdmember) {
		this.federalTaxIdmember = federalTaxIdmember;
	}

	public double getMonthly_payment() {
		return monthly_payment;
	}

	public void setMonthly_payment(double monthly_payment) {
		this.monthly_payment = monthly_payment;
	}

}
